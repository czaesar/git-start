git init иницилиазируем репозиторий, создается системная папка .git

git status текущий статус

git add "название файла"   -добавление файл в git 

git rm -- cached "название файла" убираем файл

git commit - m "название коммита" добавление коммита

git branch - ветка в которой я есть

git branch "название ветки" - создаем ветку

git branch -D "название ветки" - удаляет ветку

git checkout "название ветки" - переключаемся на ветку
